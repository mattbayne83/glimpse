## Phase 5: Web Worker for Pyodide 🚀 (P0)

### Current Problem
- Pyodide runs on main thread
- Long analyses block UI (10-15 seconds for large files)
- No progress feedback
- Browser shows "Page Unresponsive" warning

### Target Solution
- Pyodide runs in dedicated Web Worker
- Main thread stays responsive
- Show progress indicator during analysis
- Can cancel long-running analyses

### Architecture

```
Main Thread                    Web Worker
├─ UI Components               ├─ Pyodide Runtime
├─ User Interactions           ├─ Pandas Analysis
├─ Progress Display            ├─ Data Processing
└─ Result Rendering            └─ JSON Serialization
       │                              │
       ├─ postMessage ────────────────>
       │  (csvText)                    │
       │                               │
       <──────────────── postMessage ──┤
          (analysisResult)
```

### Implementation Steps

1. **Create `src/workers/pyodide.worker.ts`**
   ```typescript
   import { getPyodide } from '../utils/pyodide';
   import { analyzeData } from '../utils/analyzeData';

   // Message types
   type WorkerMessage =
     | { type: 'INIT' }
     | { type: 'ANALYZE', payload: { csvText: string } };

   type WorkerResponse =
     | { type: 'READY' }
     | { type: 'PROGRESS', payload: { percent: number, message: string } }
     | { type: 'RESULT', payload: AnalysisResult }
     | { type: 'ERROR', payload: { message: string } };

   let pyodideInstance: any = null;

   self.onmessage = async (e: MessageEvent<WorkerMessage>) => {
     const { type, payload } = e.data;

     switch (type) {
       case 'INIT':
         try {
           self.postMessage({ type: 'PROGRESS', payload: { percent: 0, message: 'Loading Python runtime...' }});
           pyodideInstance = await getPyodide();
           self.postMessage({ type: 'READY' });
         } catch (error) {
           self.postMessage({ type: 'ERROR', payload: { message: String(error) }});
         }
         break;

       case 'ANALYZE':
         try {
           self.postMessage({ type: 'PROGRESS', payload: { percent: 50, message: 'Analyzing data...' }});
           const result = await analyzeData(payload.csvText);
           self.postMessage({ type: 'RESULT', payload: result });
         } catch (error) {
           self.postMessage({ type: 'ERROR', payload: { message: String(error) }});
         }
         break;
     }
   };
   ```

2. **Create `src/utils/useWorker.ts` (Hook)**
   ```typescript
   import { useRef, useCallback } from 'react';
   import type { AnalysisResult } from '../types/analysis';

   type WorkerStatus = 'idle' | 'initializing' | 'ready' | 'analyzing' | 'error';

   interface UseWorkerReturn {
     status: WorkerStatus;
     progress: { percent: number; message: string };
     analyze: (csvText: string) => Promise<AnalysisResult>;
     terminate: () => void;
   }

   export function useWorker(): UseWorkerReturn {
     const workerRef = useRef<Worker | null>(null);
     const [status, setStatus] = useState<WorkerStatus>('idle');
     const [progress, setProgress] = useState({ percent: 0, message: '' });

     // Initialize worker
     const init = useCallback(() => {
       if (!workerRef.current) {
         workerRef.current = new Worker(
           new URL('../workers/pyodide.worker.ts', import.meta.url),
           { type: 'module' }
         );

         workerRef.current.postMessage({ type: 'INIT' });
         setStatus('initializing');
       }
     }, []);

     // Analyze function
     const analyze = useCallback((csvText: string): Promise<AnalysisResult> => {
       return new Promise((resolve, reject) => {
         if (!workerRef.current) {
           init();
         }

         const worker = workerRef.current!;

         worker.onmessage = (e) => {
           const { type, payload } = e.data;

           switch (type) {
             case 'READY':
               setStatus('ready');
               worker.postMessage({ type: 'ANALYZE', payload: { csvText } });
               break;
             case 'PROGRESS':
               setProgress(payload);
               setStatus('analyzing');
               break;
             case 'RESULT':
               setStatus('ready');
               resolve(payload);
               break;
             case 'ERROR':
               setStatus('error');
               reject(new Error(payload.message));
               break;
           }
         };
       });
     }, [init]);

     // Cleanup
     const terminate = useCallback(() => {
       workerRef.current?.terminate();
       workerRef.current = null;
       setStatus('idle');
     }, []);

     return { status, progress, analyze, terminate };
   }
   ```
